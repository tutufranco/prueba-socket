class PassengerClient {

    this.socket = io('ws://localhost:3000/events');
    this.setupEventListeners();
  }

    // Recibir datos iniciales del viaje
    this.socket.on('get-trip-p-on', (tripData) => {

    });

    // Recibir cambios de estado
    this.socket.on('send-change-trip', (tripChange) => {
      console.log('üîÑ Estado del viaje:', tripChange);
      this.handleTripChange(tripChange);
    });

    // Recibir ubicaci√≥n del conductor
    this.socket.on('driver-location-update', (location) => {
      console.log('üìç Ubicaci√≥n del conductor:', location);
      this.updateDriverLocation(location);
    });

    // Recibir incidentes
    this.socket.on('trip-incident-p-on', (incident) => {
      console.log('‚ö†Ô∏è Incidente reportado:', incident);
      this.handleIncident(incident);
    });

    // Recibir mensajes
    this.socket.on('trip-message-p-on', (message) => {
      console.log('üí¨ Mensaje recibido:', message);
      this.handleMessage(message);
    });
  }

  // Enviar ubicaci√≥n

      message: message

## üéØ MEJORES PR√ÅCTICAS

### 1. Reconexi√≥n Autom√°tica
```javascript
class WebSocketClient {
  constructor(url) {
    this.url = url;
    this.socket = null;
    this.reconnectAttempts = 0;
    this.maxReconnectAttempts = 5;
    this.connect();
  }

  connect() {
    this.socket = io(this.url);
    
    this.socket.on('connect', () => {
      console.log('Conectado');
      this.reconnectAttempts = 0;
    });

    this.socket.on('disconnect', () => {
      console.log('Desconectado');
      this.handleReconnect();
    });
  }

  handleReconnect() {
    if (this.reconnectAttempts < this.maxReconnectAttempts) {
      this.reconnectAttempts++;
      setTimeout(() => {
        console.log(`Reintentando conexi√≥n (${this.reconnectAttempts}/${this.maxReconnectAttempts})`);
        this.connect();
      }, 5000 * this.reconnectAttempts);
    }
  }
}
```

### 2. Manejo de Estado
```javascript
class TripStateManager {
  constructor() {
    this.currentState = null;
    this.listeners = [];
  }

  updateState(newState) {
    this.currentState = newState;
    this.notifyListeners(newState);
  }

  addListener(callback) {
    this.listeners.push(callback);
  }

  notifyListeners(state) {
    this.listeners.forEach(callback => callback(state));
  }
}
```
